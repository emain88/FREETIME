    <script>
        function updateClock() {
            const now = new Date();
            
            // Zeit formatieren
            const hours = String(now.getHours()).padStart(2, '0');
            const minutes = String(now.getMinutes()).padStart(2, '0');
            const seconds = String(now.getSeconds()).padStart(2, '0');
            
            // Datum formatieren
            const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
            const dateString = now.toLocaleDateString('de-DE', options);
            
            // Uhr aktualisieren
            document.getElementById('time').innerHTML = `${hours}:${minutes}:${seconds}`;
            document.getElementById('date').innerHTML = dateString;
            
            // Jede Sekunde aktualisieren
            setTimeout(updateClock, 1000);
        }
        
        // 20-Minuten Timer Funktionalität
        const TIMER_DURATION = 20 * 60; // 20 Minuten in Sekunden
        let timer;
        let timeLeft = TIMER_DURATION;
        let isRunning = false;
        
        const taskInput = document.getElementById('task-input');
        const startBtn = document.getElementById('start-btn');
        const pauseBtn = document.getElementById('pause-btn');
        const resetBtn = document.getElementById('reset-btn');
        const timerDisplay = document.getElementById('timer');
        const activeTask = document.getElementById('active-task');
        const currentTask = document.getElementById('current-task');
        const controls = document.querySelector('.controls');
        const progressContainer = document.getElementById('progress-container');
        const progressBar = document.getElementById('progress-bar');
        
        function updateTimerDisplay() {
            const minutes = Math.floor(timeLeft / 60);
            const seconds = timeLeft % 60;
            timerDisplay.textContent = `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
            
            // Update progress bar
            const progressPercentage = (TIMER_DURATION - timeLeft) / TIMER_DURATION * 100;
            progressBar.style.transform = `scaleX(${progressPercentage / 100})`;
        }
        
        function startTimer() {
            if (!isRunning && taskInput.value.trim() !== '') {
                isRunning = true;
                currentTask.textContent = taskInput.value;
                activeTask.style.visibility = 'visible';
                timerDisplay.style.visibility = 'visible';
                controls.style.visibility = 'visible';
                progressContainer.style.visibility = 'visible';
                startBtn.disabled = true;
                startBtn.style.opacity = '0.5';
                
                timer = setInterval(() => {
                    timeLeft--;
                    updateTimerDisplay();
                    
                    if (timeLeft <= 0) {
                        clearInterval(timer);
                        isRunning = false;
                        
                        // Benachrichtigung, wenn möglich
                        if ('Notification' in window && Notification.permission === 'granted') {
                            new Notification('TimeFlow', {
                                body: `20 Minuten Fokuszeit für "${taskInput.value}" abgeschlossen!`
                            });
                        }
                        
                        setTimeout(() => {
                            resetTimer();
                        }, 3000);
                    }
                }, 1000);
            }
        }
        
        function pauseTimer() {
            if (isRunning) {
                clearInterval(timer);
                isRunning = false;
                pauseBtn.textContent = 'Fortsetzen';
            } else {
                startTimer();
                pauseBtn.textContent = 'Pause';
            }
        }
        
        function resetTimer() {
            clearInterval(timer);
            isRunning = false;
            timeLeft = TIMER_DURATION;
            updateTimerDisplay();
            pauseBtn.textContent = 'Pause';
            taskInput.value = '';
            activeTask.style.visibility = 'hidden';
            timerDisplay.style.visibility = 'hidden';
            controls.style.visibility = 'hidden';
            progressContainer.style.visibility = 'hidden';
            startBtn.disabled = false;
            startBtn.style.opacity = '1';
        }
        
        startBtn.addEventListener('click', startTimer);
        pauseBtn.addEventListener('click', pauseTimer);
        resetBtn.addEventListener('click', resetTimer);
        
        // Benachrichtigungen anfordern
        if ('Notification' in window) {
            Notification.requestPermission();
        }
        
        // Initialisieren
        updateClock();
        updateTimerDisplay();
    </script>
